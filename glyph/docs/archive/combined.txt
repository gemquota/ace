import os
import random
import uuid
import copy
import sqlite3
import networkx as nx
import chromadb
import litellm
from sentence_transformers import SentenceTransformer
from .models import GLYPHS, Node, AgentPayload

litellm.success_callback = []

class Engine:
    def __init__(self, graph, seed=42):
        self.graph = graph
        self.seed = seed
        self.random = random.Random(seed)
        
        self.events = []
        self.state = {'tick': 0}
        self.history = []
        
        # ML Substrate
        self.chroma = chromadb.Client()
        self.collection = self.chroma.get_or_create_collection(name=f"swarm_{seed}")
        self.encoder = SentenceTransformer('all-MiniLM-L6-v2')
        
        # Telemetry DB
        self.db = sqlite3.connect("metrics.db", check_same_thread=False)
        self._init_db()

        self.active_nodes = [(graph.root, self.create_genesis_payload())] if graph.root else []

    def _init_db(self):
        c = self.db.cursor()
        c.execute('''CREATE TABLE IF NOT EXISTS telemetry (tick INT, agent_id TEXT, role TEXT, score REAL, entropy REAL)''')
        self.db.commit()

    def create_genesis_payload(self):
        return AgentPayload(
            current_role="Explorer",
            entropy_index=1.0,
            priority_score=1.0
        )

    def _garbage_collection(self):
        current_tick = self.state['tick']
        to_remove = []
        for nid in list(self.graph.nx_graph.nodes):
            if nid == self.graph.root: continue
            last_visit = self.graph.node_data[nid]['last_visited_tick']
            if current_tick - last_visit > 150:
                to_remove.append(nid)
        
        for nid in to_remove:
            self.graph.nx_graph.remove_node(nid)
            del self.graph.node_data[nid]
            self.events.append({"type": "sys_gc", "node": nid})

    def _call_llm(self, messages, tier, role):
        model = "gemini/gemini-1.5-flash" if tier == 1 else "gemini/gemini-1.5-pro-latest"
        temp = 1.2 if role == "Explorer" else 0.2
        try:
            return litellm.completion(model=model, messages=messages, temperature=temp).choices[0].message.content.strip()
        except Exception:
            return "[COGNITIVE FAULT]"

    def step(self):
        self.state['tick'] += 1
        next_active = []
        
        self._garbage_collection()
        
        for node_id, payload in self.active_nodes:
            if node_id not in self.graph.nx_graph: continue
            
            # Track visits
            self.graph.node_data[node_id]['visits'] += 1
            self.graph.node_data[node_id]['last_visited_tick'] = self.state['tick']
            
            # Record Telemetry
            c = self.db.cursor()
            c.execute("INSERT INTO telemetry VALUES (?, ?, ?, ?, ?)", 
                      (self.state['tick'], payload.agent_id, payload.current_role, payload.priority_score, payload.entropy_index))
            self.db.commit()

            processed_payloads = self.execute_node(node_id, payload)
            
            successors = list(self.graph.nx_graph.successors(node_id))
            for target_id in successors:
                for p in processed_payloads:
                    next_active.append((target_id, p.model_copy(deep=True)))

        self.active_nodes = next_active
        if not self.active_nodes and self.graph.root:
            self.events.append({"type": "fallback", "desc": "Graph starved. Re-anchoring."})
            self.active_nodes = [(self.graph.root, self.create_genesis_payload())]

    def execute_node(self, node_id, payload: AgentPayload):
        g = self.graph.node_data[node_id]['glyph']
        out = [payload]
        self.events.append({"type": GLYPHS.get(g, 'exec'), "node": node_id, "glyph": g})
        
        # Tier 1 (Fast Routing)
        if g == '⟡': 
            text = " ".join(payload.context_window[-2:])
            res = self._call_llm([{"role": "user", "content": f"Advance: {text}"}], tier=1, role=payload.current_role)
            payload.context_window.append(res)
            
        # Tier 2 (Complex Logic)
        elif g == '⎋': 
            payload.current_role = "Critic"
            payload.entropy_index = 0.1
            text = " ".join(payload.context_window[-3:])
            res = self._call_llm([{"role": "user", "content": f"Critique logically: {text}"}], tier=2, role=payload.current_role)
            payload.context_window.append(f"[CRITIQUE]: {res}")

        elif g == '∇': 
            payload.current_role = "Synthesizer"
            text = " ".join(payload.context_window)
            res = self._call_llm([{"role": "user", "content": f"Compress to directive: {text}"}], tier=2, role=payload.current_role)
            payload.context_window = [f"[SYNTHESIS]: {res}"]

        # Vector Memory with Quarantine
        elif g == '≈': 
            if payload.context_window:
                emb = self.encoder.encode([payload.context_window[-1]])[0].tolist()
                results = self.collection.query(query_embeddings=[emb], n_results=1, where={"coherence": {"$gte": 0.3}})
                if results['documents'] and results['documents'][0]:
                    payload.context_window.append(f"[RECALLED]: {results['documents'][0][0]}")

        elif g == '⊕': 
            text = " ".join(payload.context_window)
            if text:
                emb = self.encoder.encode([text])[0].tolist()
                doc_id = f"mem_{payload.agent_id}_{self.state['tick']}"
                self.collection.add(ids=[doc_id], embeddings=[emb], documents=[text], metadatas=[{"coherence": payload.priority_score}])

        # Topological Mutation (Guarded)
        elif g == '✂':
            nodes = list(self.graph.nx_graph.nodes)
            if len(nodes) > 1:
                target = self.random.choice(nodes)
                if target != self.graph.root and target != node_id:
                    self.graph.nx_graph.remove_node(target)
                    if self.graph.is_fractured():
                        self.graph.nx_graph.add_node(target) # Rollback
                    else:
                        del self.graph.node_data[target]

        return out

    def get_state(self):
        active_edges = []
        for nid, p in self.active_nodes:
             successors = list(self.graph.nx_graph.successors(nid))
             for s in successors:
                 active_edges.append((nid, s))

        return {
            "graph": self.graph.serialize(),
            "active_nodes": [nid for nid, p in self.active_nodes],
            "active_edges": active_edges,
            "payloads": {nid: p.model_dump() for nid, p in self.active_nodes},
            "events": self.events[-30:],
            "state": self.state
        }

import uuid
import networkx as nx
from pydantic import BaseModel, Field
from typing import List, Dict, Optional

class AgentPayload(BaseModel):
    agent_id: str = Field(default_factory=lambda: f"agt_{uuid.uuid4().hex[:4]}")
    context_window: List[str] = []
    current_role: str = "Explorer"
    priority_score: float = 1.0
    entropy_index: float = 1.0
    lineage_hash: Optional[str] = None
    memory_refs: List[str] = []
    metadata: Dict = {}

class Node:
    def __init__(self, node_id, glyph, type_name):
        self.id = node_id
        self.glyph = glyph
        self.type = type_name
        self.links = []
        self.visits = 0 
        self.last_visited_tick = 0

GLYPHS = {
    '⟐': 'anchor', '⊢': 'emerge', '⊣': 'resolve', '⟳': 'loop', '⌁': 'trace',
    '◊': 'branch', '≠': 'divergence', '✳': 'spawn', '✂': 'delete', '⇄': 'rewire', '⌬': 'mutate',
    '(': 'scope_open', ')': 'scope_close',
    '⟡': 'transform', '∇': 'compress', '꩜': 'brainstorm', '⎋': 'critique',
    '⊕': 'commit', '≈': 'recall', '⇥': 'prune',
    '⧉': 'replicate', '⛖': 'sync', '☍': 'debate', '∿': 'broadcast',
    '☁': 'uplink', '🖧': 'localize'
}

class Graph:
    def __init__(self):
        self.nx_graph = nx.DiGraph()
        self.root = None
        self.node_data = {}

    def add_node(self, node_id, glyph):
        if not self.root: self.root = node_id
        type_name = GLYPHS.get(glyph, 'unknown')
        self.nx_graph.add_node(node_id, glyph=glyph, type=type_name)
        self.node_data[node_id] = {'id': node_id, 'glyph': glyph, 'type': type_name, 'links': [], 'visits': 0, 'last_visited_tick': 0}

    def add_edge(self, src, tgt):
        if src in self.nx_graph and tgt in self.nx_graph:
            self.nx_graph.add_edge(src, tgt)

    def is_fractured(self):
        if len(self.nx_graph.nodes) == 0: return False
        components = list(nx.weakly_connected_components(self.nx_graph))
        return len(components) > 1

    def serialize(self):
        nodes = []
        for nid in self.nx_graph.nodes:
            data = self.node_data[nid]
            data['links'] = list(self.nx_graph.successors(nid))
            nodes.append(data)
        return {'nodes': nodes, 'root': self.root}

    def load_serialized(self, data):
        self.nx_graph.clear()
        self.node_data.clear()
        self.root = data.get('root')
        for n in data.get('nodes', []):
            self.add_node(n['id'], n['glyph'])
        for n in data.get('nodes', []):
            for target in n.get('links', []):
                self.add_edge(n['id'], target)

def parse_gfs(text):
    graph = Graph()
    stack, last_node_id = [], None
    for char in text:
        if char in GLYPHS and char not in ['(', ')']:
            nid = str(uuid.uuid4())[:8]
            graph.add_node(nid, char)
            if stack: graph.add_edge(stack[-1], nid)
            elif last_node_id: graph.add_edge(last_node_id, nid)
            last_node_id = nid
        elif char == '(':
            if last_node_id: stack.append(last_node_id)
            last_node_id = None
        elif char == ')':
            if stack: last_node_id = stack.pop()
    return graph

export const API = {
    async request(url, options = {}) {
        try {
            const response = await fetch(url, options);
            if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
            return await response.json();
        } catch (e) {
            console.error(`API Error at ${url}:`, e);
            return null;
        }
    },
    loadProgram: (text) => API.request('/api/load', { 
        method: 'POST', 
        headers: {'Content-Type': 'application/json'}, 
        body: JSON.stringify({text}) 
    }),
    play: () => API.request('/api/play', { method: 'POST' }),
    pause: () => fetch('/api/pause', { method: 'POST' })
};

export class Renderer {
    constructor(canvas) {
        this.canvas = canvas;
        this.ctx = canvas.getContext('2d');
        this.nodePositions = {};
        this.resize();
        window.addEventListener('resize', () => this.resize());
    }
    
    resize() {
        const parent = this.canvas.parentElement;
        if(parent) {
            this.canvas.width = parent.clientWidth;
            this.canvas.height = parent.clientHeight;
        }
    }
    
    updatePhysics(stateData) {
        if(!stateData || !stateData.graph) return;
        const nodes = stateData.graph.nodes;
        const cw = this.canvas.width;
        const ch = this.canvas.height;
        
        nodes.forEach(n => {
            if(!this.nodePositions[n.id]) {
                this.nodePositions[n.id] = { x: cw/2 + (Math.random()-0.5)*50, y: ch/2 + (Math.random()-0.5)*50, vx: 0, vy: 0 };
            }
        });
        
        const repel = 500;
        for(let i=0; i<nodes.length; i++) {
            for(let j=i+1; j<nodes.length; j++) {
                let p1 = this.nodePositions[nodes[i].id];
                let p2 = this.nodePositions[nodes[j].id];
                let dx = p1.x - p2.x;
                let dy = p1.y - p2.y;
                let dist = Math.sqrt(dx*dx + dy*dy) || 1;
                if(dist < 100) {
                    let f = repel / (dist * dist);
                    p1.vx += (dx/dist)*f; p1.vy += (dy/dist)*f;
                    p2.vx -= (dx/dist)*f; p2.vy -= (dy/dist)*f;
                }
            }
        }

        nodes.forEach(n => {
            let p1 = this.nodePositions[n.id];
            (n.links || []).forEach(targetId => {
                let p2 = this.nodePositions[targetId];
                if(p2) {
                    let dx = p2.x - p1.x;
                    let dy = p2.y - p1.y;
                    let dist = Math.sqrt(dx*dx + dy*dy) || 1;
                    let f = (dist - 80) * 0.05;
                    p1.vx += (dx/dist)*f; p1.vy += (dy/dist)*f;
                    p2.vx -= (dx/dist)*f; p2.vy -= (dy/dist)*f;
                }
            });
            p1.vx += (cw/2 - p1.x) * 0.005;
            p1.vy += (ch/2 - p1.y) * 0.005;
            p1.vx *= 0.85; p1.vy *= 0.85;
            p1.x += p1.vx; p1.y += p1.vy;
        });
    }
    
    draw(stateData) {
        this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
        if(!stateData || !stateData.graph) return;
        const nodes = stateData.graph.nodes;
        const activeNodes = stateData.active_nodes || [];
        const activeEdges = stateData.active_edges || [];
        
        this.ctx.lineWidth = 2;
        nodes.forEach(n => {
            let p1 = this.nodePositions[n.id];
            if(!p1) return;
            (n.links || []).forEach(targetId => {
                let p2 = this.nodePositions[targetId];
                if(p2) {
                    // Highlight active traversal edges in #ff0055
                    let isActiveEdge = activeEdges.some(e => e[0] === n.id && e[1] === targetId);
                    this.ctx.strokeStyle = isActiveEdge ? '#ff0055' : '#445566';
                    
                    this.ctx.beginPath();
                    this.ctx.moveTo(p1.x, p1.y);
                    this.ctx.lineTo(p2.x, p2.y);
                    this.ctx.stroke();
                }
            });
        });

        nodes.forEach(n => {
            let p = this.nodePositions[n.id];
            if(!p) return;
            this.ctx.fillStyle = activeNodes.includes(n.id) ? '#ff0055' : '#112233';
            this.ctx.strokeStyle = '#00ffcc';
            this.ctx.beginPath();
            this.ctx.arc(p.x, p.y, 15, 0, Math.PI*2);
            this.ctx.fill();
            this.ctx.stroke();
            this.ctx.fillStyle = '#ffffff';
            this.ctx.font = '14px monospace';
            this.ctx.textAlign = 'center';
            this.ctx.textBaseline = 'middle';
            this.ctx.fillText(n.glyph, p.x, p.y);
        });
    }
}

import { API } from './api.js';

export class Builder {
    constructor(onUpdate) {
        this.onUpdate = onUpdate;
        this.program = [];
        this.loadLibrary();
    }
    
    updateContext() {
        this.onUpdate(this.program.join(''));
    }
    
    async loadLibrary() {
        try {
            const lib = await API.getLibrary();
            const container = document.getElementById('library-programs');
            if(!container || !lib) return;
            container.innerHTML = '';
            
            for(let [name, code] of Object.entries(lib)) {
                let btn = document.createElement('button');
                btn.className = 'full-width-btn';
                btn.textContent = name;
                btn.onclick = () => {
                    const input = document.getElementById('program-input');
                    if(input) input.value = code;
                    const drawer = document.getElementById('library-drawer');
                    if(drawer) drawer.classList.add('collapsed');
                };
                container.appendChild(btn);
            }
        } catch (e) {
            console.error("Library extraction failed:", e);
        }
    }
}

    bind('btn-play', API.play);
    bind('btn-pause', API.pause);
    bind('btn-load', () => {
        const input = document.getElementById('program-input');
        if(!input) return;
        API.pause().then(() => {
            API.loadProgram(input.value);
            renderer.nodePositions = {}; 
        });
    });

    // 4. WebSocket Telemetry
    const ws = new WebSocket(`ws://${window.location.host}/ws`);
    ws.onmessage = (event) => {
        lastGraphState = JSON.parse(event.data);
        const view = document.getElementById('state-view');
        if(view && lastGraphState.state) {
            view.textContent = `Tick: ${lastGraphState.state.tick} | Nodes: ${lastGraphState.graph.nodes.length}`;
        }
    };

    function loop() {
        if (lastGraphState) {
            renderer.updatePhysics(lastGraphState);
            renderer.draw(lastGraphState);
        }
        requestAnimationFrame(loop);
    }
    requestAnimationFrame(loop);

} catch (err) {
    // Fallback alert ensures you see the stack trace through Termux/VNC
    alert("JAVASCRIPT CRASH: " + err.message);
    console.error(err);
}

# Nerd Fonts

This is an archived font from the Nerd Fonts release v3.3.0.

For more information see:
* https://github.com/ryanoasis/nerd-fonts/
* https://github.com/ryanoasis/nerd-fonts/releases/latest/

# Meslo LG

Meslo LG is a customized version of Apple’s Menlo-Regular font
(which is a customized Bitstream Vera Sans Mono).

We use a patched version by opeik https://github.com/opeik here.

For more information have a look at the upstream website: https://github.com/andreberg/Meslo-Font

Version: 1.21

## Which font?

### TL;DR

* Pick your font family:
  * If you are limited to monospaced fonts (because of your terminal, etc) then pick a font with `Nerd Font Mono` (or `NFM`).
  * If you want to have bigger icons (usually around 1.5 normal letters wide) pick a font without `Mono` i.e. `Nerd Font` (or `NF`). Most terminals support this, but ymmv.
  * If you work in a proportional context (GUI elements or edit a presentation etc) pick a font with `Nerd Font Propo` (or `NFP`).

### Ligatures

Ligatures are generally preserved in the patched fonts.
Nerd Fonts `v2.0.0` had no ligatures in the `Nerd Font Mono` fonts, this has been dropped with `v2.1.0`.
If you have a ligature-aware terminal and don't want ligatures you can (usually) disable them in the terminal settings.

### Explanation

Once you narrow down your font choice of family (`Droid Sans`, `Inconsolata`, etc) and style (`bold`, `italic`, etc) you have 2 main choices:

#### `Option 1: Download already patched font`

 * For a stable version download a font package from the [release page](https://github.com/ryanoasis/nerd-fonts/releases)
 * Or download the development version from the folders here

#### `Option 2: Patch your own font`

 * Patch your own variations with the various options provided by the font patcher (i.e. not include all symbols for smaller font size)

For more information see: [The FAQ](https://github.com/ryanoasis/nerd-fonts/wiki/FAQ-and-Troubleshooting#which-font)

import os
import random
import uuid
import copy
import numpy as np
import chromadb
import tiktoken
from sentence_transformers import SentenceTransformer
import litellm
from .models import GLYPHS, Node

# Prevent LiteLLM from attempting to log to external telemetry platforms
litellm.success_callback = []

class Engine:
    def __init__(self, graph, seed=42):
        self.graph = graph
        self.seed = seed
        self.random = random.Random(seed)
        
        # Temporal and State Arrays
        self.events = []
        self.state = {'tick': 0, 'global_memory': {}}
        self.history = []
        self.holding_cells = {} 
        
        print("Initializing ML Substrate (ChromaDB, Sentence-Transformers, TikToken)...")
        # Initialize persistent local vector database
        self.chroma = chromadb.Client()
        self.collection = self.chroma.get_or_create_collection(name=f"swarm_{seed}")
        
        # Initialize local dense embedding model for semantic search
        self.encoder = SentenceTransformer('all-MiniLM-L6-v2')
        self.tokenizer = tiktoken.get_encoding("cl100k_base")
        
        self.active_nodes = [(graph.root, self.create_genesis_payload())] if graph.root else []

    def create_genesis_payload(self):
        """Initializes a pristine agent payload at the root anchor."""
        return {
            "agent_id": f"agt_{str(uuid.uuid4())[:4]}",
            "context_window": ["Agent spawned."],
            "hardware_route": "cloud", # Defaults to Gemini API
            "entropy": 1.0
        }

    def save_snapshot(self):
        """Generates a fully decoupled deep-copy of the dimensional graph state."""
        return {
            "graph_data": self.graph.serialize(),
            "active_nodes": copy.deepcopy(self.active_nodes),
            "events": copy.deepcopy(self.events),
            "state": copy.deepcopy(self.state),
            "holding_cells": copy.deepcopy(self.holding_cells)
        }

    def load_snapshot(self, snap):
        """Restores the execution matrix from a historical snapshot."""
        self.graph.load_serialized(snap["graph_data"])
        self.active_nodes = copy.deepcopy(snap["active_nodes"])
        self.events = copy.deepcopy(snap["events"])
        self.state = copy.deepcopy(snap["state"])
        self.holding_cells = copy.deepcopy(snap.get("holding_cells", {}))

    def step(self):
        """Advances the entire topological swarm by one discrete microsecond."""
        # Anchor the timeline before mutating
        self.history.append(self.save_snapshot())
        self.state['tick'] += 1
        next_active = []
        
        for node_id, payload in self.active_nodes:
            if node_id not in self.graph.nx_graph: continue
            
            g = self.graph.node_data[node_id]['glyph']
            
            # --- SWARM SYNC BARRIER LOGIC ---
            if g == '⛖': 
                if node_id not in self.holding_cells: self.holding_cells[node_id] = []
                self.holding_cells[node_id].append(payload)
                # Trap the agent if a peer has not arrived at the node
                if len(self.holding_cells[node_id]) < 2:
                    self.events.append({"type": "sync_wait", "node": node_id, "agent": payload["agent_id"]})
                    next_active.append((node_id, payload)) 
                    continue
                else:
                    # Release swarm lock
                    self.events.append({"type": "sync_release", "node": node_id})
                    self.holding_cells[node_id] = []

            # Execute standard physics & neural logic
            processed_payloads = self.execute_node(node_id, payload)
            
            successors = list(self.graph.nx_graph.successors(node_id))
            for target_id in successors:
                for p in processed_payloads:
                    next_active.append((target_id, dict(p)))

        self.active_nodes = next_active
        if not self.active_nodes and self.graph.root:
            self.events.append({"type": "fallback", "desc": "Graph starved. Re-anchoring."})
            self.active_nodes = [(self.graph.root, self.create_genesis_payload())]

    def step_back(self):
        """Pops the execution stack to move backwards in time."""
        if self.history:
            previous_state = self.history.pop()
            self.load_snapshot(previous_state)
            return True
        return False

    def _call_llm(self, messages, route, temperature=0.7):
        """Directs the cognitive load to the defined hardware API."""
        if route == "local":
            # Points to the local llama.cpp server running inside the Termux / Proot layer
            return litellm.completion(
                model="openai/local-model", 
                api_base="http://127.0.0.1:8081/v1",
                api_key="sk-local",
                messages=messages,
                temperature=temperature
            ).choices[0].message.content.strip()
        else:
            # Reaches out to the official Google Gemini API endpoint
            return litellm.completion(
                model="gemini/gemini-1.5-flash",
                messages=messages,
                temperature=temperature
            ).choices[0].message.content.strip()

    def execute_node(self, node_id, payload):
        """Executes the specific behavior bound to the active glyph."""
        g = self.graph.node_data[node_id]['glyph']
        out = [payload]
        self.events.append({"type": GLYPHS.get(g, 'exec'), "node": node_id, "glyph": g})
        route = payload.get("hardware_route", "cloud")
        
        # --- HARDWARE ROUTING ---
        if g == '☁': payload['hardware_route'] = "cloud"
        elif g == '🖧': payload['hardware_route'] = "local"
        
        # --- COGNITIVE PIPELINE ---
        elif g == '⟡': 
            text = " ".join(payload['context_window'][-3:])
            try:
                res = self._call_llm([{"role": "user", "content": f"Advance this thought process one step: {text}"}], route)
                payload['context_window'].append(res)
            except Exception as e: payload['context_window'].append("[LLM FAULT]")

        elif g == '꩜': 
            text = " ".join(payload['context_window'][-2:])
            try:
                res = self._call_llm([{"role": "user", "content": f"Generate a wildly creative divergence from this: {text}"}], route, temperature=1.2)
                payload['context_window'].append(f"[DIVERGENCE]: {res}")
            except Exception: pass

        elif g == '∇': 
            text = " ".join(payload['context_window'])
            try:
                res = self._call_llm([{"role": "user", "content": f"Summarize to a dense core directive: {text}"}], route)
                payload['context_window'] = [f"[SYNTHESIS]: {res}"]
            except Exception: pass

        elif g == '⎋': 
            text = " ".join(payload['context_window'][-2:])
            try:
                res = self._call_llm([{"role": "user", "content": f"Identify a logical flaw or blindspot in this thought: {text}"}], route)
                payload['context_window'].append(f"[CRITIQUE]: {res}")
            except Exception: pass

        # --- MULTI-AGENT SWARM DYNAMICS ---
        elif g == '⊢': 
            payload['agent_id'] = f"agt_{str(uuid.uuid4())[:4]}"
            payload['context_window'].append(f"Branched context at tick {self.state['tick']}")

        elif g == '⧉': 
            clone = copy.deepcopy(payload)
            clone['agent_id'] = f"{payload['agent_id']}_clone"
            out.append(clone) # Outputs two payload instances 
            
        elif g == '∿': 
            msg = payload['context_window'][-1]
            self.events.append({"type": "broadcast", "msg": msg})
            for nid, other_payload in self.active_nodes:
                if other_payload['agent_id'] != payload['agent_id']:
                    other_payload['context_window'].append(f"[BROADCAST from {payload['agent_id']}]: {msg}")

        elif g == '☍': 
            peers = [p for nid, p in self.active_nodes if nid == node_id and p['agent_id'] != payload['agent_id']]
            if peers:
                peer = peers[0]
                text1 = " ".join(payload['context_window'][-2:])
                text2 = " ".join(peer['context_window'][-2:])
                try:
                    res = self._call_llm([{"role": "user", "content": f"Synthesize a resolution between these contexts. A: {text1}. B: {text2}"}], route)
                    payload['context_window'].append(f"[RESOLUTION]: {res}")
                    peer['context_window'].append(f"[RESOLUTION]: {res}")
                except Exception: pass

        # --- CHROMADB VECTOR MEMORY ---
        elif g == '≈': 
            try:
                emb = self.encoder.encode([payload['context_window'][-1]])[0].tolist()
                results = self.collection.query(query_embeddings=[emb], n_results=1)
                if results['documents'] and results['documents'][0]:
                    payload['context_window'].append(f"[RECALLED]: {results['documents'][0][0]}")
            except Exception: pass
            
        elif g == '⊕': 
            text = " ".join(payload['context_window'])
            try:
                emb = self.encoder.encode([text])[0].tolist()
                doc_id = f"mem_{payload['agent_id']}_{self.state['tick']}"
                self.collection.add(ids=[doc_id], embeddings=[emb], documents=[text])
                payload['context_window'] = [f"Archived to Hivemind: {doc_id}"]
            except Exception: pass

        elif g == '⇥': 
            if len(payload['context_window']) > 2:
                payload['context_window'].pop(0) 
                payload['context_window'].append("[PRUNED OLDEST MEMORY]")

        # --- TOPOLOGICAL MUTATIONS ---
        elif g == '✳':
            new_id = str(uuid.uuid4())[:8]
            self.graph.add_node(new_id, self.random.choice([k for k in GLYPHS.keys() if k not in ['(',')']]))
            self.graph.add_edge(node_id, new_id)
            
        elif g == '⌬':
            self.graph.node_data[node_id]['glyph'] = self.random.choice([k for k in GLYPHS.keys() if k not in ['(',')']])
            self.graph.node_data[node_id]['type'] = GLYPHS.get(self.graph.node_data[node_id]['glyph'], 'unknown')

        elif g == '✂':
            nodes = list(self.graph.nx_graph.nodes)
            if len(nodes) > 1:
                target = self.random.choice(nodes)
                if target != self.graph.root and target != node_id:
                    self.graph.nx_graph.remove_node(target)
                    del self.graph.node_data[target]

        elif g == '⇄':
            nodes = list(self.graph.nx_graph.nodes)
            if len(nodes) > 1:
                target = self.random.choice(nodes)
                self.graph.add_edge(node_id, target)

        elif g == '⤴':
            successors = list(self.graph.nx_graph.successors(node_id))
            if successors:
                target_id = self.random.choice(successors)
                orig_glyph = self.graph.node_data[target_id]['glyph']
                new_id = str(uuid.uuid4())[:8]
                self.graph.add_node(new_id, orig_glyph)
                self.graph.add_edge(node_id, new_id)

        elif g == '≠':
            successors = list(self.graph.nx_graph.successors(node_id))
            if successors:
                target_id = self.random.choice(successors)
                self.graph.nx_graph.remove_edge(node_id, target_id)

        return out

    def get_state(self):
        return {
            "graph": self.graph.serialize(),
            "active": [nid for nid, p in self.active_nodes],
            "payloads": {nid: p for nid, p in self.active_nodes},
            "events": self.events[-30:],
            "state": self.state,
            "text": self.graph.to_text(),
            "seed": self.seed
        }

import uuid
import networkx as nx

# The missing piece the engine was looking for
class Node:
    def __init__(self, node_id, glyph, type_name):
        self.id = node_id
        self.glyph = glyph
        self.type = type_name
        self.links = []

GLYPHS = {
    # Core & Topological
    '⟐': 'anchor', '⊢': 'emerge', '⊣': 'resolve', '⟳': 'loop', '⌁': 'trace',
    '◊': 'branch', '≠': 'divergence', '✳': 'spawn', '✂': 'delete', '⇄': 'rewire', '⌬': 'mutate',
    '(': 'scope_open', ')': 'scope_close',
    # Cognitive
    '⟡': 'transform', '∇': 'compress', '꩜': 'brainstorm', '⎋': 'critique',
    # Vector Memory
    '⊕': 'commit', '≈': 'recall', '⇥': 'prune',
    # Swarm Dynamics
    '⧉': 'replicate', '⛖': 'sync', '☍': 'debate', '∿': 'broadcast',
    # Hardware Routing
    '☁': 'uplink', '🖧': 'localize'
}

class Graph:
    def __init__(self):
        self.nx_graph = nx.DiGraph()
        self.root = None
        self.node_data = {}

    def add_node(self, node_id, glyph):
        if not self.root: self.root = node_id
        type_name = GLYPHS.get(glyph, 'unknown')
        self.nx_graph.add_node(node_id, glyph=glyph, type=type_name)
        # Store as a dictionary for easier serialization
        self.node_data[node_id] = {'id': node_id, 'glyph': glyph, 'type': type_name, 'links': []}

    def add_edge(self, src, tgt):
        if src in self.nx_graph and tgt in self.nx_graph:
            self.nx_graph.add_edge(src, tgt)

    def serialize(self):
        nodes = []
        for nid in self.nx_graph.nodes:
            data = self.node_data[nid]
            data['links'] = list(self.nx_graph.successors(nid))
            nodes.append(data)
        return {'nodes': nodes, 'root': self.root}

    def load_serialized(self, data):
        self.nx_graph.clear()
        self.node_data.clear()
        self.root = data.get('root')
        for n in data.get('nodes', []):
            self.add_node(n['id'], n['glyph'])
        for n in data.get('nodes', []):
            for target in n.get('links', []):
                self.add_edge(n['id'], target)

    def to_text(self):
        if not self.root: return ""
        visited, out = set(), []
        def traverse(nid):
            if nid in visited or nid not in self.nx_graph: return
            visited.add(nid)
            out.append(self.node_data[nid]['glyph'])
            successors = list(self.nx_graph.successors(nid))
            if successors:
                out.append('(')
                for child in successors: traverse(child)
                out.append(')')
        traverse(self.root)
        return "".join(out)

def parse_gfs(text):
    graph = Graph()
    stack, last_node_id = [], None
    for char in text:
        if char in GLYPHS and char not in ['(', ')']:
            nid = str(uuid.uuid4())[:8]
            graph.add_node(nid, char)
            if stack: graph.add_edge(stack[-1], nid)
            elif last_node_id: graph.add_edge(last_node_id, nid)
            last_node_id = nid
        elif char == '(':
            if last_node_id: stack.append(last_node_id); last_node_id = None
        elif char == ')':
            if stack: last_node_id = stack.pop()
    return graph

export const API = {
    async request(url, options = {}) {
        try {
            const response = await fetch(url, options);
            if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
            return await response.json();
        } catch (e) {
            console.error(`API Error at ${url}:`, e);
            return null;
        }
    },
    getState: () => API.request('/api/state'),
    getLibrary: () => API.request('/api/library'),
    saveToLibrary: (name, text) => API.request('/api/library', { 
        method: 'POST', 
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({name, text}) 
    }),
    loadProgram: (text) => API.request('/api/load', { 
        method: 'POST', 
        headers: {'Content-Type': 'application/json'}, 
        body: JSON.stringify({text}) 
    }),
    play: () => API.request('/api/play', { method: 'POST' }),
    pause: () => fetch('/api/pause', { method: 'POST' }),
    step: () => API.request('/api/step', { method: 'POST'}),
    stepBack: () => API.request('/api/step_back', { method: 'POST'}),
    getTrace: () => API.request('/api/trace'),
    restoreSnapshot: (snapshot) => API.request('/api/restore_snapshot', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify(snapshot)
    })
};

export class Renderer {
    constructor(canvas) {
        this.canvas = canvas;
        this.ctx = canvas.getContext('2d');
        this.nodePositions = {};
        this.resize();
        window.addEventListener('resize', () => this.resize());
    }
    resize() {
        const parent = this.canvas.parentElement;
        if(parent) {
            this.canvas.width = parent.clientWidth;
            this.canvas.height = parent.clientHeight;
        }
    }
    updatePhysics(stateData) {
        if(!stateData || !stateData.graph) return;
        const nodes = stateData.graph.nodes;
        const cw = this.canvas.width;
        const ch = this.canvas.height;
        
        nodes.forEach(n => {
            if(!this.nodePositions[n.id]) {
                this.nodePositions[n.id] = { x: cw/2 + (Math.random()-0.5)*50, y: ch/2 + (Math.random()-0.5)*50, vx: 0, vy: 0 };
            }
        });

        const repel = 500;
        for(let i=0; i<nodes.length; i++) {
            for(let j=i+1; j<nodes.length; j++) {
                let p1 = this.nodePositions[nodes[i].id];
                let p2 = this.nodePositions[nodes[j].id];
                let dx = p1.x - p2.x;
                let dy = p1.y - p2.y;
                let dist = Math.sqrt(dx*dx + dy*dy) || 1;
                if(dist < 100) {
                    let f = repel / (dist * dist);
                    p1.vx += (dx/dist)*f; p1.vy += (dy/dist)*f;
                    p2.vx -= (dx/dist)*f; p2.vy -= (dy/dist)*f;
                }
            }
        }

        nodes.forEach(n => {
            let p1 = this.nodePositions[n.id];
            (n.links || []).forEach(targetId => {
                let p2 = this.nodePositions[targetId];
                if(p2) {
                    let dx = p2.x - p1.x;
                    let dy = p2.y - p1.y;
                    let dist = Math.sqrt(dx*dx + dy*dy) || 1;
                    let f = (dist - 80) * 0.05;
                    p1.vx += (dx/dist)*f; p1.vy += (dy/dist)*f;
                    p2.vx -= (dx/dist)*f; p2.vy -= (dy/dist)*f;
                }
            });
            p1.vx += (cw/2 - p1.x) * 0.005;
            p1.vy += (ch/2 - p1.y) * 0.005;
            p1.vx *= 0.85; p1.vy *= 0.85;
            p1.x += p1.vx; p1.y += p1.vy;
        });
    }
    draw(stateData) {
        this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
        if(!stateData || !stateData.graph) return;
        const nodes = stateData.graph.nodes;
        
        this.ctx.strokeStyle = '#445566';
        this.ctx.lineWidth = 2;
        nodes.forEach(n => {
            let p1 = this.nodePositions[n.id];
            if(!p1) return;
            (n.links || []).forEach(targetId => {
                let p2 = this.nodePositions[targetId];
                if(p2) {
                    this.ctx.beginPath();
                    this.ctx.moveTo(p1.x, p1.y);
                    this.ctx.lineTo(p2.x, p2.y);
                    this.ctx.stroke();
                }
            });
        });

        nodes.forEach(n => {
            let p = this.nodePositions[n.id];
            if(!p) return;
            this.ctx.fillStyle = '#112233';
            this.ctx.strokeStyle = '#00ffcc';
            this.ctx.beginPath();
            this.ctx.arc(p.x, p.y, 15, 0, Math.PI*2);
            this.ctx.fill();
            this.ctx.stroke();
            this.ctx.fillStyle = '#ffffff';
            this.ctx.font = '14px monospace';
            this.ctx.textAlign = 'center';
            this.ctx.textBaseline = 'middle';
            this.ctx.fillText(n.glyph, p.x, p.y);
        });
    }
}

import { API } from './api.js';

export class Builder {
    constructor(onUpdate) {
        this.onUpdate = onUpdate;
        this.program = [];
        this.loadLibrary();
    }
    updateContext() {
        this.onUpdate(this.program.join(''));
    }
    async loadLibrary() {
        try {
            const lib = await API.getLibrary();
            const container = document.getElementById('library-list');
            if(!container || !lib) return;
            container.innerHTML = '';
            for(let [name, code] of Object.entries(lib)) {
                let btn = document.createElement('button');
                btn.className = 'full-width-btn';
                btn.textContent = name;
                btn.onclick = () => {
                    const input = document.getElementById('program-input');
                    if(input) input.value = code;
                    const drawer = document.getElementById('library-drawer');
                    if(drawer) drawer.classList.add('collapsed');
                };
                container.appendChild(btn);
            }
        } catch (e) {
            console.error("Library failed to load:", e);
        }
    }
}

import { API } from './api.js';
import { Renderer } from './canvas.js';
import { Builder } from './editor.js';

try {
    const canvasEl = document.getElementById('canvas');
    if (!canvasEl) throw new Error("CRITICAL: <canvas id='canvas'> not found!");
    const renderer = new Renderer(canvasEl);

    let lastGraphState = null;
    const DEFAULT_PROG = "⟐(⊢◊(≈⟡⊣⌁)(≠✳⌬⇄)⊕)⟳";

    const builder = new Builder((text) => {
        const input = document.getElementById('program-input');
        if (input) input.value = text;
    });

    const bind = (id, fn) => {
        const el = document.getElementById(id);
        if (el) el.onclick = fn;
        else console.warn(`Warning: Button '${id}' missing.`);
    };

    bind('btn-play', API.play);
    bind('btn-pause', API.pause);
    bind('btn-step', API.step);
    bind('btn-step-back', API.stepBack);
    bind('btn-reset', () => {
        API.pause().then(() => {
            API.loadProgram(DEFAULT_PROG);
            const input = document.getElementById('program-input');
            if (input) input.value = DEFAULT_PROG;
            renderer.nodePositions = {};
            if(builder) {
                builder.program = DEFAULT_PROG.split('');
                builder.updateContext();
            }
        });
    });

    bind('btn-load', () => {
        const input = document.getElementById('program-input');
        if(!input) return alert("Missing program-input ID!");
        API.pause().then(() => {
            API.loadProgram(input.value);
            renderer.nodePositions = {}; 
        });
    });

    bind('builder-toggle', () => {
        const drawer = document.getElementById('builder-drawer');
        const icon = document.getElementById('builder-icon');
        if (drawer) {
            drawer.classList.toggle('collapsed');
            if (icon) icon.textContent = drawer.classList.contains('collapsed') ? '↑' : '↓';
            setTimeout(() => renderer.resize(), 310);
        }
    });

    bind('btn-toggle-lib', () => {
        const lib = document.getElementById('library-drawer');
        if (lib) lib.classList.remove('collapsed');
    });

    bind('btn-close-lib', () => {
        const lib = document.getElementById('library-drawer');
        if (lib) lib.classList.add('collapsed');
    });

    setInterval(async () => {
        try {
            lastGraphState = await API.getState();
            const view = document.getElementById('state-view');
            if(lastGraphState && lastGraphState.state && view) {
                view.textContent = `Tick: ${lastGraphState.state.tick} | Nodes: ${lastGraphState.graph.nodes.length}`;
            }
        } catch (e) { /* Poll fail safe */ }
    }, 500);

    function loop() {
        if (lastGraphState) {
            renderer.updatePhysics(lastGraphState);
            renderer.draw(lastGraphState);
        }
        requestAnimationFrame(loop);
    }
    requestAnimationFrame(loop);

    console.log("GFS-II Client Initialized Successfully");

} catch (err) {
    alert("JAVASCRIPT CRASH: " + err.message);
    console.error(err);
}

            '☁': { type: 'uplink', desc: 'Hardware: Route to Cloud API.', validNext: ['⊢','⟡','('] },
            '🖧': { type: 'localize', desc: 'Hardware: Route to Local LLM.', validNext: ['⊢','⟡','('] }

        const API = {
            async request(url, options = {}) {
                try {
                    const res = await fetch(url, options);
                    if (!res.ok) throw new Error();

                } catch (e) { return null; }
            },
            getState: () => API.request('/api/state'),
            getLibrary: () => API.request('/api/library'),
            saveToLibrary: (name, text) => API.request('/api/library', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({name, text}) }),
            loadProgram: (text) => API.request('/api/load', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({text}) }),
            play: () => API.request('/api/play', { method: 'POST' }),
            pause: () => fetch('/api/pause', { method: 'POST' }),
            step: () => API.request('/api/step', { method: 'POST'}),
            stepBack: () => API.request('/api/step_back', { method: 'POST'}),
            getTrace: () => API.request('/api/trace')

        // --- 3. UI Modules ---
        class Builder {
            constructor() {
                this.program = [];
                this.activeGlyph = null;
                
                this.uiPalette = document.getElementById('glyph-palette');
                this.uiDetails = document.getElementById('glyph-details');
                this.uiTrack = document.getElementById('builder-track');
                this.uiInput = document.getElementById('program-input');
                
                document.getElementById('btn-add-glyph').onclick = () => this.addCurrent();
                document.getElementById('btn-rm-glyph').onclick = () => this.removeLast();
                document.getElementById('btn-save-lib').onclick = () => this.save();
                
                this.renderPalette();
                this.updateContext();
                this.loadLibrary();
            }
            renderPalette() {
                this.uiPalette.innerHTML = '';
                const validNext = this.getValidNext();
                for (const [g, data] of Object.entries(DICTIONARY)) {
                    const btn = document.createElement('button');
                    btn.textContent = g;
                    if (this.program.length === 0 && g !== '⟐') btn.classList.add('faded');
                    else if (this.program.length > 0 && !validNext.includes(g)) btn.classList.add('faded');
                    btn.onclick = () => this.selectGlyph(g, data);
                    this.uiPalette.appendChild(btn);
                }
            }
            getValidNext() {
                if (this.program.length === 0) return ['⟐'];
                const last = this.program[this.program.length - 1];
                return DICTIONARY[last]?.validNext || Object.keys(DICTIONARY);
            }
            selectGlyph(g, data) {
                this.activeGlyph = g;
                this.uiDetails.innerHTML = `
                    <h4 style="margin-bottom:5px; color:#fff">${g} - <span style="color:var(--accent)">${data.type.toUpperCase()}</span></h4>
                    <p style="margin-bottom:8px">${data.desc}</p>
                    <small style="color:var(--text-dim)">Valid Links: ${data.validNext.join(' ')}</small>
                `;
            }
            addCurrent() {
                if (!this.activeGlyph) return;
                this.program.push(this.activeGlyph);
                this.updateContext();
            }
            removeLast() {
                this.program.pop();
                this.updateContext();
            }
            updateContext() {
                const progStr = this.program.join('');
                this.uiTrack.textContent = progStr || 'Empty';
                this.uiInput.value = progStr;
                this.renderPalette();
            }

        class LogViewer {
            constructor() {
                this.container = document.getElementById('log-output');
                this.lastTick = -1;
                this.knownPayloads = {};
            }
            update(stateData) {
                if (!stateData || !stateData.state) return;
                const currentTick = stateData.state.tick;
                
                if (currentTick !== this.lastTick && currentTick >= 0) {
                    this.lastTick = currentTick;
                    
                    // 1. Log System Events
                    if(stateData.events) {
                        stateData.events.forEach(e => {
                            if(e.type === 'exec') return; // Too noisy
                            this.print(`<span class="log-tick">[T:${currentTick}]</span> <span class="log-sys">[SYS]</span> ${e.type.toUpperCase()} at ${e.node.substring(0,6)}`);
                        });
                    }

                    // 2. Diff Context Windows to find LLM Packets
                    if(stateData.payloads) {
                        for (let [nid, payload] of Object.entries(stateData.payloads)) {
                            const aid = payload.agent_id || 'unknown';
                            const context = payload.context_window || [];
                            const prev = this.knownPayloads[aid] || [];
                            
                            if (context.length > prev.length) {
                                const newItems = context.slice(prev.length);
                                newItems.forEach(item => {
                                    this.print(`<span class="log-tick">[T:${currentTick}]</span> <span class="log-llm">[PACKET: ${aid}]</span> ${item}`);
                                });
                            }
                            this.knownPayloads[aid] = [...context];
                        }
                    }
                }
            }
            print(htmlStr) {
                const div = document.createElement('div');
                div.className = 'log-entry';
                div.innerHTML = htmlStr;
                this.container.appendChild(div);
                this.container.scrollTop = this.container.scrollHeight;
            }
            clear() {
                this.container.innerHTML = '';
                this.lastTick = -1;
                this.knownPayloads = {};
            }
        }

        class Renderer {
            constructor(canvas) {
                this.canvas = canvas; this.ctx = canvas.getContext('2d'); this.nodePositions = {};
                this.resize(); window.addEventListener('resize', () => this.resize());
            }
            resize() {
                const parent = this.canvas.parentElement;
                if(parent) { this.canvas.width = parent.clientWidth; this.canvas.height = parent.clientHeight; }
            }
            updatePhysics(stateData) {
                if(!stateData || !stateData.graph) return;
                const nodes = stateData.graph.nodes;
                const cw = this.canvas.width; const ch = this.canvas.height;
                nodes.forEach(n => { if(!this.nodePositions[n.id]) this.nodePositions[n.id] = { x: cw/2 + (Math.random()-0.5)*50, y: ch/2 + (Math.random()-0.5)*50, vx: 0, vy: 0 }; });
                const repel = 500;
                for(let i=0; i<nodes.length; i++) {
                    for(let j=i+1; j<nodes.length; j++) {
                        let p1 = this.nodePositions[nodes[i].id]; let p2 = this.nodePositions[nodes[j].id];
                        let dx = p1.x - p2.x; let dy = p1.y - p2.y; let dist = Math.sqrt(dx*dx + dy*dy) || 1;
                        if(dist < 100) { let f = repel / (dist * dist); p1.vx += (dx/dist)*f; p1.vy += (dy/dist)*f; p2.vx -= (dx/dist)*f; p2.vy -= (dy/dist)*f; }
                    }
                }
                nodes.forEach(n => {
                    let p1 = this.nodePositions[n.id];
                    (n.links || []).forEach(targetId => {
                        let p2 = this.nodePositions[targetId];
                        if(p2) {
                            let dx = p2.x - p1.x; let dy = p2.y - p1.y; let dist = Math.sqrt(dx*dx + dy*dy) || 1;
                            let f = (dist - 80) * 0.05; p1.vx += (dx/dist)*f; p1.vy += (dy/dist)*f; p2.vx -= (dx/dist)*f; p2.vy -= (dy/dist)*f;
                        }
                    });
                    p1.vx += (cw/2 - p1.x) * 0.005; p1.vy += (ch/2 - p1.y) * 0.005;
                    p1.vx *= 0.85; p1.vy *= 0.85; p1.x += p1.vx; p1.y += p1.vy;
                });
            }
            draw(stateData) {
                this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
                if(!stateData || !stateData.graph) return;
                const nodes = stateData.graph.nodes;
                const activeIds = stateData.active || [];
                
                this.ctx.strokeStyle = '#445566'; this.ctx.lineWidth = 2;
                nodes.forEach(n => {
                    let p1 = this.nodePositions[n.id];
                    if(!p1) return;
                    (n.links || []).forEach(targetId => {
                        let p2 = this.nodePositions[targetId];
                        if(p2) { this.ctx.beginPath(); this.ctx.moveTo(p1.x, p1.y); this.ctx.lineTo(p2.x, p2.y); this.ctx.stroke(); }
                    });
                });

                nodes.forEach(n => {
                    let p = this.nodePositions[n.id];
                    if(!p) return;
                    this.ctx.fillStyle = activeIds.includes(n.id) ? '#ff0055' : '#112233';
                    this.ctx.strokeStyle = '#00ffcc';
                    this.ctx.beginPath(); this.ctx.arc(p.x, p.y, 15, 0, Math.PI*2); this.ctx.fill(); this.ctx.stroke();
                    this.ctx.fillStyle = '#ffffff'; this.ctx.font = '14px monospace'; this.ctx.textAlign = 'center'; this.ctx.textBaseline = 'middle';
                    this.ctx.fillText(n.glyph, p.x, p.y);
                });
            }
        }

        // --- 4. Main Initialization ---
        try {
            const renderer = new Renderer(document.getElementById('canvas'));
            const builder = new Builder();
            const logViewer = new LogViewer();
            let lastGraphState = null;

            // Bindings
            const bind = (id, fn) => { const el = document.getElementById(id); if (el) el.onclick = fn; };
            const toggleDrawer = (id) => document.getElementById(id).classList.toggle('open');
            
            bind('btn-toggle-builder', () => toggleDrawer('builder-drawer'));
            bind('btn-close-builder', () => toggleDrawer('builder-drawer'));
            bind('btn-toggle-lib', () => toggleDrawer('library-drawer'));
            bind('btn-close-lib', () => toggleDrawer('library-drawer'));
            bind('btn-toggle-log', () => toggleDrawer('log-drawer'));

            bind('btn-play', API.play);
            bind('btn-pause', API.pause);
            bind('btn-step-back', async () => {
                await API.pause();
                await API.stepBack();
                // Because we stepped back, the log viewer needs a hard reset to avoid duplicate packet detection
                logViewer.clear(); 
            });
            bind('btn-step', async () => {
                document.getElementById('state-view').textContent = "THINKING (Processing Nodes)...";
                await API.step();
            });
            
            // Reload injects the current string fresh, resetting physics and logs
            bind('btn-reload', async () => {
                await API.pause();
                const text = document.getElementById('program-input').value;
                await API.loadProgram(text);
                renderer.nodePositions = {};
                logViewer.clear();
                document.getElementById('state-view').textContent = "Tick: 0 | Nodes: 0";
            });
            
            bind('btn-load', () => document.getElementById('btn-reload').click());

            bind('btn-export-trace', async () => {
                await API.pause();
                const trace = await API.getTrace();
                const blob = new Blob([JSON.stringify(trace, null, 2)], {type: 'application/json'});
                const a = document.createElement('a'); a.href = URL.createObjectURL(blob);
                a.download = `gfs_trace_${Date.now()}.json`; a.click();
            });

            // Polling Loop
            setInterval(async () => {
                lastGraphState = await API.getState();
                const view = document.getElementById('state-view');
                if(lastGraphState && lastGraphState.state && view && !view.textContent.includes("THINKING")) {
                    view.textContent = `Tick: ${lastGraphState.state.tick} | Nodes: ${lastGraphState.graph.nodes.length}`;
                    logViewer.update(lastGraphState);
                }
            }, 500);

            // Render Loop
            function loop() {
                if (lastGraphState) { renderer.updatePhysics(lastGraphState); renderer.draw(lastGraphState); }
                requestAnimationFrame(loop);
            }
            requestAnimationFrame(loop);

        } catch (err) { alert("CRASH: " + err.message); }
    </script>

# Meslo LG

Meslo LG is a customized version of Apple’s Menlo-Regular font
(which is a customized Bitstream Vera Sans Mono).

We use a patched version by opeik https://github.com/opeik here.

For more information have a look at the upstream website: https://github.com/andreberg/Meslo-Font

Version: 1.21

## Which font?

### TL;DR

* Pick your font family:
  * If you are limited to monospaced fonts (because of your terminal, etc) then pick a font with `Nerd Font Mono` (or `NFM`).
  * If you want to have bigger icons (usually around 1.5 normal letters wide) pick a font without `Mono` i.e. `Nerd Font` (or `NF`). Most terminals support this, but ymmv.
  * If you work in a proportional context (GUI elements or edit a presentation etc) pick a font with `Nerd Font Propo` (or `NFP`).

### Ligatures

Ligatures are generally preserved in the patched fonts.
Nerd Fonts `v2.0.0` had no ligatures in the `Nerd Font Mono` fonts, this has been dropped with `v2.1.0`.
If you have a ligature-aware terminal and don't want ligatures you can (usually) disable them in the terminal settings.

### Explanation

Once you narrow down your font choice of family (`Droid Sans`, `Inconsolata`, etc) and style (`bold`, `italic`, etc) you have 2 main choices:

#### `Option 1: Download already patched font`

 * For a stable version download a font package from the [release page](https://github.com/ryanoasis/nerd-fonts/releases)
 * Or download the development version from the folders here

#### `Option 2: Patch your own font`

 * Patch your own variations with the various options provided by the font patcher (i.e. not include all symbols for smaller font size)

For more information see: [The FAQ](https://github.com/ryanoasis/nerd-fonts/wiki/FAQ-and-Troubleshooting#which-font)

import os
import json
import asyncio
from fastapi import FastAPI, Request
from fastapi.staticfiles import StaticFiles
from fastapi.responses import FileResponse
from pydantic import BaseModel
import uvicorn
from src.backend.models import parse_gfs
from src.backend.engine import Engine

app = FastAPI(title="GFS-II Neural IDE")

LIB_FILE = 'library/programs.json'
os.makedirs('library', exist_ok=True)
DEFAULT_PROGRAM = "⟐(⊢◊(≈⟡⊣⌁)(≠✳⌬⇄)⊕)⟳"

def seed_library():
    if not os.path.exists(LIB_FILE):
        with open(LIB_FILE, 'w') as f: 
            json.dump({"Target Loop": DEFAULT_PROGRAM}, f)
seed_library()

engine_instance = Engine(parse_gfs(DEFAULT_PROGRAM))
running = False

class LoadRequest(BaseModel):
    text: str
    seed: int = 42

@app.get("/")
async def read_index(): return FileResponse("web/index.html")

@app.get("/api/state")
async def get_state(): return engine_instance.get_state()

@app.post("/api/step")
async def step_engine():
    # Threading bypass: Allows the server to stay alive while the LLM thinks
    loop = asyncio.get_event_loop()
    await loop.run_in_executor(None, engine_instance.step)
    return {"status": "ok"}

@app.post("/api/play")
async def play_engine():
    global running
    running = True
    return {"status": "running"}

@app.post("/api/pause")
async def pause_engine():
    global running
    running = False
    return {"status": "paused"}

@app.post("/api/load")
async def load_engine(req: LoadRequest):
    global running, engine_instance
    running = False
    engine_instance = Engine(parse_gfs(req.text), seed=req.seed)
    return {"status": "ok"}

@app.get("/api/library")
async def get_library():
    with open(LIB_FILE, 'r') as f: return json.load(f)

async def engine_daemon():
    global running, engine_instance
    loop = asyncio.get_event_loop()
    while True:
        if running and engine_instance:
            try:
                # Run the heavy ML computation in a separate thread
                await loop.run_in_executor(None, engine_instance.step)
            except Exception as e:
                print(f"Engine Step Error: {e}")
                running = False
        await asyncio.sleep(0.5)

@app.on_event("startup")
async def startup_event():
    asyncio.create_task(engine_daemon())

if os.path.exists("web"):
    app.mount("/web", StaticFiles(directory="web"), name="web")

if __name__ == "__main__":
    uvicorn.run("main:app", host="0.0.0.0", port=8080, reload=True)

# GFS-II: Reflexive Field System

A pocket-sized, enterprise-grade runtime where language = graph = runtime. 

## Overview
GFS-II is a unified symbolic computation substrate. The system operates entirely on Unicode glyphs representing nodes in an activation-based executable graph. It naturally supports self-modification, live trace viewing, and direct visual mapping.

## Structure
- `gfs.py`: The Python backend, containing the core Execution Engine, Graph structure, textual/graph parsing logic, and the HTTP Server for the IDE.
- `web/`: The Canvas-based visual IDE for real-time monitoring and manipulation.

## Core Glyphs
- **⟐** anchor (start)
- **⊢** emerge (task init)
- **⟡** transform (task logic)
- **⊣** resolve (task end)
- **⌁** trace (log)
- **⟳** loop (reactivate)

## Mutation Glyphs
- **✳** spawn (create node)
- **✂** delete (remove node)
- **⇄** rewire (change edges)
- **⌬** mutate (change glyph)
- **⤴** clone (duplicate node)
- **⤵** compress (remove node between)

## Usage

### CLI execution
```sh
# Run minimal target program
./gfs.py run

# Load specific file
./gfs.py run examples/target.gfs

# Step through explicitly
./gfs.py step
```

### Visual IDE
```sh
# Start web server on port 8080
./gfs.py serve
```
Open your browser to `http://localhost:8080/web/index.html` to view the live graph.

# GFS-II: Reflexive Field System
A modular, self-modifying, graph-based visual IDE and execution runtime. GFS-II abandons traditional linear execution paradigms, operating entirely on Unicode glyphs representing nodes in an activation-based network. It natively supports self-modification, live matrix-based pan/zoom rendering, physics-based topology layout, and visual execution tracing.
Built with a philosophy that aligns with practical evolutionary computation, GFS-II blurs the line between static data and living, mutating processes. It is designed for environments where dynamic adaptation is valued over rigid, unchanging principles.
## Architectural Philosophy
Traditional code executes top-down. GFS-II executes topologically. Programs are not scripts; they are physical spaces where logic flows like electrical impulses through a neural substrate.
 * **Reflexivity:** The graph can read, spawn, mutate, and delete its own nodes at runtime.
 * **Spatial Execution:** Control flow is dictated by graph edges, allowing concurrent divergence and unpredictable convergence.
 * **Practical Fluidity:** The system prioritizes immediate execution and runtime adaptation over compile-time strictness.
## Directory & Module Structure
The system has been completely modularized to separate the physics/execution engine, the REST API, and the visual frontend. This ensures practical expandability and clean separation of concerns.
.
├── server.py                 # Core ASGI/HTTP server and API router. Connects UI to Engine.
├── library/
│   └── programs.json         # Persistent JSON storage for saved graphs.
├── src/

│       └── models.py         # Graph structures, node definitions, and the string-to-GFS parser.
└── web/
├── index.html            # Main UI layout and viewport definitions.
├── style.css             # UI styling for the editor and canvas.

└── editor.js         # Syntax rules, state machine, palette rendering, and builder UI.
## Core Engine Mechanics
GFS-II evaluates a graph by maintaining a dynamic list of active_nodes. Time in the system is measured in discrete "Ticks."
### The Execution Lifecycle
During each tick, the engine performs the following deterministic sequence:
 1. **Read Active State:** Snapshots the current array of active node IDs.
 2. **Execute Glyphs:** Iterates over the snapshot. For each node, it executes the specific mathematical or topological behavior bound to that node's glyph.
 3. **State Mutation:** Applies any graph-altering commands (e.g., node deletion, creation, or rewiring) generated by the executed glyphs.
 4. **Edge Propagation:** Moves the "active" status from the currently executing nodes to all valid linked child nodes via connecting edges.
 5. **Root Fallback:** If the active node pool empties (a dead end), the engine automatically attempts to re-anchor at the graph's designated root node.
### Physics and Topology
The graphical representation is not just cosmetic; it utilizes a repulsive/attractive force-directed graph physics model. Nodes repel each other to maintain visual clarity, while edges act as springs pulling connected logic blocks together.
## Glyph Syntax & Dictionary
The system uses specific glyphs to dictate control flow, state mutation, and graph topography. The visual editor enforces valid topological connections based on these rules, preventing fatal structural errors before execution begins.
### Foundational Control Flow
 * **⟐ (Anchor):** Anchored starting state. Serves as the immutable root of the program.
 * **⊢ (Emerge):** A new task emerges into active context. Often used to initialize a distinct processing branch.

These glyphs alter the physical structure of the program while it is running, allowing for self-optimizing or chaotic growth.

#!/usr/bin/env python3
import os, json, time, threading
from http.server import SimpleHTTPRequestHandler, HTTPServer
from src.backend.models import parse_gfs
from src.backend.engine import Engine

LIB_FILE = 'library/programs.json'
if not os.path.exists('library'): os.makedirs('library')
if not os.path.exists(LIB_FILE):
    with open(LIB_FILE, 'w') as f: json.dump({"Target Loop": "⟐(⊢◊(≈⟡⊣⌁)(≠✳⌬⇄)⊕)⟳"}, f)

engine_instance = Engine(parse_gfs("⟐⟳"))
running = False

class WebHandler(SimpleHTTPRequestHandler):
    def do_GET(self):
        if self.path == '/': self.path = '/web/index.html'
        if self.path == '/api/state':
            self._send_json(engine_instance.get_state())
            return
        if self.path == '/api/library':
            with open(LIB_FILE, 'r') as f: self._send_json(json.load(f))
            return
        return super().do_GET()

    def do_POST(self):
        global running, engine_instance
        content_length = int(self.headers.get('Content-Length', 0))
        data = json.loads(self.rfile.read(content_length)) if content_length > 0 else {}

        if self.path == '/api/step':
            engine_instance.step()
            self._send_json({"status": "ok"})
        elif self.path == '/api/play':
            running = True
            self._send_json({"status": "running"})
        elif self.path == '/api/pause':
            running = False
            self._send_json({"status": "paused"})
        elif self.path == '/api/load':
            running = False
            engine_instance = Engine(parse_gfs(data.get('text', '⟐⟳')), seed=int(data.get('seed', 42)))
            self._send_json({"status": "ok"})
        elif self.path == '/api/library':
            with open(LIB_FILE, 'r') as f: lib = json.load(f)
            lib[data.get('name', 'Untitled')] = data.get('text')
            with open(LIB_FILE, 'w') as f: json.dump(lib, f)
            self._send_json({"status": "saved"})

    def _send_json(self, obj):
        self.send_response(200)
        self.send_header('Content-type', 'application/json')
        self.end_headers()
        self.wfile.write(json.dumps(obj).encode())

def engine_loop():
    global running
    while True:
        if running and engine_instance: engine_instance.step()
        time.sleep(0.5)

if __name__ == "__main__":
    t = threading.Thread(target=engine_loop, daemon=True)
    t.start()
    print("🚀 API & UI Server active at http://localhost:8080/web/index.html")
    HTTPServer(('0.0.0.0', 8080), WebHandler).serve_forever()

    if not os.path.exists(LIB_FILE):
        with open(LIB_FILE, 'w') as f:

seed_library()

engine_instance = Engine(parse_gfs(DEFAULT_PROGRAM))
running = False

    text: str
    seed: int = 42

async def read_index(): return FileResponse("web/index.html")

async def get_state(): return engine_instance.get_state()

@app.post("/api/step")
async def step_engine():
    # Threading bypass: Allows the server to stay alive while the LLM thinks
    loop = asyncio.get_event_loop()
    await loop.run_in_executor(None, engine_instance.step)
    return {"status": "ok"}

@app.post("/api/play")
async def play_engine():
    global running
    running = True
    return {"status": "running"}

@app.post("/api/pause")
async def pause_engine():
    global running
    running = False
    return {"status": "paused"}

@app.post("/api/load")
async def load_engine(req: LoadRequest):
    global running, engine_instance
    running = False
    engine_instance = Engine(parse_gfs(req.text), seed=req.seed)
    return {"status": "ok"}

@app.get("/api/library")
async def get_library():
    with open(LIB_FILE, 'r') as f: return json.load(f)

async def engine_daemon():
    global running, engine_instance
    loop = asyncio.get_event_loop()
    while True:
        if running and engine_instance:
            try:
                # Run the heavy ML computation in a separate thread
                await loop.run_in_executor(None, engine_instance.step)
            except Exception as e:
                print(f"Engine Step Error: {e}")
                running = False
        await asyncio.sleep(0.5)

@app.on_event("startup")
async def startup_event():
    asyncio.create_task(engine_daemon())

if os.path.exists("web"):
    app.mount("/web", StaticFiles(directory="web"), name="web")

if __name__ == "__main__":
    uvicorn.run("main:app", host="0.0.0.0", port=8080, reload=True)

## Core Engine Mechanics
GFS-II evaluates a graph by maintaining a dynamic list of active_nodes. Time in the system is measured in discrete "Ticks."
### The Execution Lifecycle
During each tick, the engine performs the following deterministic sequence:
 1. **Read Active State:** Snapshots the current array of active node IDs.
 2. **Execute Glyphs:** Iterates over the snapshot. For each node, it executes the specific mathematical or topological behavior bound to that node's glyph.
 3. **State Mutation:** Applies any graph-altering commands (e.g., node deletion, creation, or rewiring) generated by the executed glyphs.
 4. **Edge Propagation:** Moves the "active" status from the currently executing nodes to all valid linked child nodes via connecting edges.
 5. **Root Fallback:** If the active node pool empties (a dead end), the engine automatically attempts to re-anchor at the graph's designated root node.

### Termux & Core System Packages
To run and develop GFS-II natively on a mobile device, the following packages provide the optimal substrate:
 * **python**: The core execution runtime for the backend engine.
 * **x11-repo & tigervnc**: Essential packages for routing the graphical Web UI to a remote viewer (like a Windows Remote Desktop client) or a local Android X11 app.
 * **termux-api**: Can be directly imported into engine.py via subprocess to trigger physical device responses. For example, executing a ✂ (Delete) node could trigger termux-vibrate to provide tactile feedback of graph destruction.
 * **proot-distro**: Highly recommended for establishing an isolated Ubuntu or Debian subsystem if complex native dependencies (like C++ graph libraries) are required later.
 * **nodejs & yarn**: For future build-chain integration, allowing you to bundle the ES6 modules in web/js/ via Vite or Webpack for a production deployment.
 * **coreutils & jq**: Indispensable for CLI-based manipulation and querying of the library/programs.json file without requiring a Python shell.
### Python PIP Libraries
The backend can be supercharged with the following Python packages to increase performance and capabilities:
 * **fastapi & uvicorn**: For upgrading the backend to a high-throughput ASGI server. If the tick-rate of massive evolutionary graphs overwhelms the standard SimpleHTTPRequestHandler, Uvicorn will handle the concurrency natively.
 * **networkx**: Replaces the basic Python dictionary graph implementation with highly optimized C-backed algorithms for graph analysis, cycle detection, and rapid pathfinding.
 * **pydantic**: Enforces strict data validation for incoming JSON payloads on the API layer, preventing malformed graphs from crashing the engine.
 * **websockets**: Allows migration from the current HTTP polling interval (400ms) to true, real-time bidirectional physics updates pushed directly from the engine to the canvas.
 * **matplotlib**: Generates static, server-side topological maps of the graph state headlessly, useful for saving execution snapshots over time.
### Model Context Protocol (MCP) Servers
For integrating GFS-II with external AI agents or larger data ecosystems:
 * **sqlite-mcp**: Facilitates the migration of library/programs.json into a structured, queriable database, logging every node execution history and mutation over the lifetime of a program.
 * **filesystem-mcp**: Enables secure, programmatic access to the host OS, allowing an AI agent to import complex external .gfs scripts or update the editor.js dictionary dynamically.
 * **github-mcp**: Allows seamless, automated version control. As the graph mutates and evolves, github-mcp can automatically push diffs of the resulting .gfs strings to a remote repository.
### Gemini CLI Extensions
For rapid, AI-assisted development within the terminal:
 * **gemini-shell**: Bridges the Termux environment to an AI context. If a volatile graph configuration causes a Python stack trace, gemini-shell allows for immediate, in-terminal diagnosis.
 * **gemini-coder**: Automates the repetitive boilerplate generation required when adding new nodes to models.py and syncing the backend dictionary with the frontend editor.js logic.
 * **gemini-git**: Provides highly contextualized commit messages, analyzing how specific topological changes to the .gfs strings alter the spatial execution flow of the programs.

@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await websocket.accept()
    connected_clients.append(websocket)
    try:
        # BUG FIX: Push the initial graph topology immediately upon connection
        await websocket.send_json(engine_instance.get_state())
        while True:
            await websocket.receive_text()
    except WebSocketDisconnect:
        connected_clients.remove(websocket)

@app.post("/api/load")
async def load_engine(req: LoadRequest):
    global running, engine_instance
    running = False
    engine_instance = Engine(parse_gfs(req.text), seed=req.seed)
    
    # BUG FIX: Force a broadcast to all clients so the canvas updates while paused
    state = engine_instance.get_state()
    for client in connected_clients:
        await client.send_json(state)
        
    return {"status": "ok"}
